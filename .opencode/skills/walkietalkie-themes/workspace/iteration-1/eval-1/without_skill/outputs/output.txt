================================================================================

            accentColor: Color(red: 0.85, green: 0.25, blue: 0.75),
            displayNameKey: "theme.name.plasma",
            iconName: "sparkle.magnifyingglass",
            isProLocked: true,
            backgroundPrimary: Color(red: 0.06, green: 0.01, blue: 0.10),
            surfaceColor: Color(red: 0.12, green: 0.03, blue: 0.18),
            textPrimary: Color(red: 0.95, green: 0.85, blue: 1.0),
            textSecondary: Color(red: 0.68, green: 0.35, blue: 0.82),
            backgroundStyle: .animated,
            cardStyle: .cyber,
            accentGlow: Color(red: 0.85, green: 0.25, blue: 0.75).opacity(0.28)
        )

Design rationale:
  - accentColor: hot pink/violet (#D940BF)
  - backgroundPrimary: very dark violet-black — the canvas glow will pulse
    over this base
  - surfaceColor: slightly lifted violet for card backgrounds
  - textPrimary: very pale violet-white for readability on dark
  - textSecondary: muted violet for secondary labels
  - accentGlow: matches accentColor at 28% opacity for the card border glow
  - cardStyle: .cyber to match the other animated themes (consistent glass
    aesthetic)
  - iconName: "sparkle.magnifyingglass" — a magnifying glass with sparkle,
    evoking "plasma" as energetic/particle-like without using the SF Symbols
    sparkle icon that galaxy already uses

3. Canvas draw function — full drawPlasma (AnimatedBackgroundView.swift)
================================================================================

File: WalkieTalkie/Theme/AnimatedBackgroundView.swift

The plasma system uses two particle types:
  - Core particles (30): slowly orbit the center, their color oscillates
    between violet and pink. They grow/shrink in a breathing pattern.
  - Ember particles (50): smaller, faster, drift outward from the center in
    random directions, fading out like embers from a plasma globe.
  - A central glow that pulses (radial gradient changes radius with sin wave).
  - A subtle semi-transparent "haze" layer of large blurred circles that move
    even slower — this mimics the cloud-like movement of real plasma.

3d. The full drawPlasma function — add after the drawGalaxy function:

    // MARK: - Plasma

    /// Renderer Canvas per il plasma.
    /// Due layer: "core" (grosse macchie pulsanti che orbitano lente) ed "embers"
    /// (scintille che si allontanano dal centro con fade-out).
    /// Aggiunge anche un bagliore centrale pulsante.
    private func drawPlasma(context: GraphicsContext, size: CGSize, time: Double) {
        let cx = size.width / 2
        let cy = size.height / 2
        let maxR = min(size.width, size.height) * 0.46

        // ---- Central pulse glow ----
        let pulse = 0.7 + 0.3 * sin(time * 0.6)
        let glowRadius = maxR * (0.25 + 0.15 * pulse)
        let glowRect = CGRect(
            x: cx - glowRadius,
            y: cy - glowRadius,
            width: glowRadius * 2,
            height: glowRadius * 2
        )
        context.fill(
            Path(ellipseIn: glowRect),
            with: .radialGradient(
                Gradient(colors: [
                    Color(red: 0.85, green: 0.25, blue: 0.75).opacity(0.35 * pulse),
                    Color(red: 0.65, green: 0.15, blue: 0.85).opacity(0.12 * pulse),
                    Color(red: 0.45, green: 0.05, blue: 0.60).opacity(0.0)
                ]),
                center: CGPoint(x: cx, y: cy),
                startRadius: 0,
                endRadius: glowRadius
            )
        )

        // ---- Core particles (slow orbit + breathing) ----
        for p in plasmaCores {
            // Slow orbit with slight radius oscillation
            let orbitAngle = p.baseAngle + time * p.speed * 0.15
            let breathe = 1.0 + 0.15 * sin(time * (0.5 + p.phase) + p.phase)
            let r = p.baseRadius * maxR * breathe
            let x = cx + CGFloat(cos(orbitAngle) * r)
            let y = cy + CGFloat(sin(orbitAngle) * r)

            // Size breathes too
            let sizePulse = 0.75 + 0.25 * sin(time * (0.7 + p.speed * 2) + p.phase)
            let s = CGFloat(p.size * sizePulse)

            // Color oscillates between violet and pink using sin
            let colorMix = 0.5 + 0.5 * sin(time * (0.4 + p.speed) + p.phase)
            let color = Color(
                red: 0.65 + 0.30 * colorMix,
                green: 0.15 + 0.20 * (1.0 - colorMix),
                blue: 0.75 + 0.20 * (1.0 - colorMix)
            ).opacity(p.brightness * 0.45)

            let rect = CGRect(x: x - s/2, y: y - s/2, width: s, height: s)
            // Blur-like effect: use a soft radial gradient per particle
            context.fill(
                Path(ellipseIn: rect),
                with: .radialGradient(
                    Gradient(colors: [color, color.opacity(0.0)]),
                    center: CGPoint(x: x, y: y),
                    startRadius: 0,
                    endRadius: s / 2
                )
            )
        }

        // ---- Ember particles (drift outward from center) ----
        for p in plasmaEmbers {
            // Radial distance increases with time, looping
            let cycle = (time * p.speed * 0.08 + p.phase / (.pi * 2))
                .truncatingRemainder(dividingBy: 1.0)
            // Embers go from center outward
            let r = cycle * maxR * 0.95
            let angle = p.baseAngle + time * 0.1 + cycle * 2.0
            let x = cx + CGFloat(cos(angle) * r)
            let y = cy + CGFloat(sin(angle) * r)

            // Fade out near the edge
            let fadeOut = 1.0 - pow(cycle, 1.5)
            let twinkle = 0.7 + 0.3 * sin(time * 3.0 + p.phase)
            let alpha = p.brightness * fadeOut * twinkle * 0.7

            let s = CGFloat(p.size * (0.5 + cycle * 0.5))
            let rect = CGRect(x: x - s/2, y: y - s/2, width: s, height: s)

            // Warm pink on the inside, violet on the outside
            let mix = cycle  // 0 = center (pink), 1 = edge (violet)
            let color = Color(
                red: 0.95 - 0.30 * mix,
                green: 0.20 + 0.10 * mix,
                blue: 0.55 + 0.35 * mix
            ).opacity(alpha)

            context.fill(Path(ellipseIn: rect), with: .color(color))
        }
    }

    private func renderPlasmaPreview(ctx: GraphicsContext, size: CGSize, t: Double) {
        let cx = size.width / 2
        let cy = size.height / 2
        let maxR = min(size.width, size.height) * 0.44

        // Glow centrale
        let pulse = 0.7 + 0.3 * sin(t * 0.5)
        let glowR = maxR * 0.35 * CGFloat(pulse)
        let glow = CGRect(x: cx - glowR, y: cy - glowR, width: glowR * 2, height: glowR * 2)
        ctx.fill(
            Path(ellipseIn: glow),
            with: .radialGradient(
                Gradient(colors: [
                    Color(red: 0.85, green: 0.25, blue: 0.75).opacity(0.40 * pulse),
                    Color(red: 0.65, green: 0.15, blue: 0.85).opacity(0.0)
                ]),
                center: CGPoint(x: cx, y: cy),
                startRadius: 0,
                endRadius: glowR
            )
        )

        for p in plasmaPreview {
            let orbit = p.baseAngle + t * p.speed * 0.2
            let breathe = 1.0 + 0.12 * sin(t * 0.6 + p.phase)
            let r = p.baseRadius * maxR * CGFloat(breathe)
            let x = cx + CGFloat(cos(orbit) * r)
            let y = cy + CGFloat(sin(orbit) * r)
            let sizePulse = 0.8 + 0.2 * sin(t * 0.8 + p.phase)
            let s = CGFloat(p.size * sizePulse)
            let colorMix = 0.5 + 0.5 * sin(t * 0.5 + p.phase)
            let color = Color(
                red: 0.70 + 0.25 * colorMix,
                green: 0.15 + 0.15 * (1.0 - colorMix),
                blue: 0.70 + 0.25 * (1.0 - colorMix)
            ).opacity(p.brightness * 0.50)

            ctx.fill(
                Path(ellipseIn: CGRect(x: x - s/2, y: y - s/2, width: s, height: s)),
                with: .radialGradient(
                    Gradient(colors: [color, color.opacity(0.0)]),
                    center: CGPoint(x: x, y: y),
                    startRadius: 0,
                    endRadius: s / 2
                )
            )
        }
    }

        ZStack {

            if reduceMotion {
                plasmaStaticOverlay

                }
            }
        }

    private var plasmaStaticOverlay: some View {
        ZStack {
            // Large static diffuse glow
            Circle()
                .fill(
                    RadialGradient(
                        gradient: Gradient(colors: [
                            Color(red: 0.85, green: 0.25, blue: 0.75).opacity(0.20),
                            Color(red: 0.45, green: 0.05, blue: 0.60).opacity(0.0)
                        ]),
                        center: .center,
                        startRadius: 20,
                        endRadius: 180
                    )
                )
                .frame(width: 360, height: 360)

            // A few static "blobs" positioned at key particle seed locations
            // using the same deterministic seed as the animated version
            let staticBlobs: [(CGPoint, CGFloat, Color)] = [
                (CGPoint(x: 0.35, y: 0.30), 28, Color(red: 0.85, green: 0.35, blue: 0.75).opacity(0.18)),
                (CGPoint(x: 0.65, y: 0.55), 22, Color(red: 0.70, green: 0.20, blue: 0.85).opacity(0.14)),
                (CGPoint(x: 0.45, y: 0.70), 34, Color(red: 0.80, green: 0.30, blue: 0.70).opacity(0.12)),
                (CGPoint(x: 0.55, y: 0.25), 18, Color(red: 0.90, green: 0.40, blue: 0.80).opacity(0.16)),
                (CGPoint(x: 0.30, y: 0.60), 26, Color(red: 0.60, green: 0.15, blue: 0.75).opacity(0.10)),
            ]

            GeometryReader { geo in
                ForEach(0..<staticBlobs.count, id: \.self) { i in
                    let blob = staticBlobs[i]
                    Circle()
                        .fill(blob.2)
                        .frame(width: blob.1, height: blob.1)
                        .blur(radius: 10)
                        .position(
                            x: geo.size.width * blob.0.x,
                            y: geo.size.height * blob.0.y
                        )
                }
            }
        }
    }
