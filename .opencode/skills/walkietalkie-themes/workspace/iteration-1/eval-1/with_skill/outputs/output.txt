====================================================================
THEME "PLASMA" — COMPLETE OUTPUT
====================================================================

Tema animato Pro-locked con particelle di plasma viola/rosa pulsanti.
Si appoggia ai pattern 3 (Onda Fluida) e 4 (Nebbia Float) del reference
gpu-shader-templates.md: combinazione di floating drift sinusoidale,
pulsazione dimensionale e glow ad anello per ogni particella.

Palette design:
- Accent: magenta-viola saturo rgb(0.85, 0.20, 0.85)
- Sfondo: viola nerissimo rgb(0.06, 0.02, 0.08)
- Surface: viola scuro rgb(0.12, 0.05, 0.15)
- Particelle: distribuzione hue 0.72–0.92 (viola → rosa → magenta)
- Card style: .cyber (effetto vetro plasma)
- Icon: sparkles
- Font: nil (system)
- Sound: nil
- Glow: 0.24

--- 3d. Computed property plasmaView ---

@ViewBuilder
private var plasmaView: some View {
    ZStack {
        // Base gradient — deep purple-black void
        LinearGradient(
            gradient: Gradient(colors: [
                Color(red: 0.05, green: 0.01, blue: 0.06),
                Color(red: 0.09, green: 0.03, blue: 0.12)
            ]),
            startPoint: .top,
            endPoint: .bottom
        )

        if reduceMotion {
            plasmaStaticOverlay
        } else {
            TimelineView(.animation(minimumInterval: 1.0 / 60.0)) { timeline in
                Canvas { context, size in
                    let t = timeline.date.timeIntervalSinceReferenceDate
                    drawPlasma(context: context, size: size, time: t)
                }
            }
        }
    }
}

--- 3e. Draw function drawPlasma(context:size:time:) ---

private func drawPlasma(context: GraphicsContext, size: CGSize, time: Double) {
    for p in plasmaParticles {
        let driftX = sin(time * p.speed * 0.3 + p.phase) * p.driftX
        let driftY = cos(time * p.speed * 0.25 + p.phase * 1.3) * p.driftY

        let x = ((p.x + driftX).truncatingRemainder(dividingBy: 1.0) + 1.0)
            .truncatingRemainder(dividingBy: 1.0) * size.width
        let y = ((p.y + driftY).truncatingRemainder(dividingBy: 1.0) + 1.0)
            .truncatingRemainder(dividingBy: 1.0) * size.height

        // Pulsing size: oscillates between 0.5x and 1.5x base size
        let pulse = 0.5 + 0.5 * sin(time * (0.6 + p.speed * 0.4) + p.phase * 2.0)
        let particleSize = CGFloat(p.size * (0.6 + pulse * 0.9))
        let glowSize = particleSize * 3.0

        // Pulsing brightness: oscillates for "breathing" plasma effect
        let breath = 0.4 + 0.6 * sin(time * (0.5 + p.speed * 0.3) + p.phase)
        let alpha = p.brightness * max(0.15, breath)

        // Generate color from hue in purple-pink-magenta range
        let color = Color(
            hue: p.hue,
            saturation: 0.8 + 0.2 * sin(time * 0.2 + p.phase),
            brightness: 0.8 + 0.2 * breath
        ).opacity(alpha)

        // Glow ring (larger, more transparent)
        let glowRect = CGRect(
            x: x - glowSize / 2, y: y - glowSize / 2,
            width: glowSize, height: glowSize
        )
        context.fill(
            Path(ellipseIn: glowRect),
            with: .color(color.opacity(alpha * 0.18))
        )

        // Core particle
        let coreRect = CGRect(
            x: x - particleSize / 2, y: y - particleSize / 2,
            width: particleSize, height: particleSize
        )
        context.fill(
            Path(ellipseIn: coreRect),
            with: .color(color)
        )
    }

    // Optional: subtle ambient glow layer
    let ambientGlow = Color(red: 0.85, green: 0.20, blue: 0.85).opacity(0.04)
    let ambientRect = CGRect(
        x: 0, y: 0,
        width: size.width, height: size.height
    )
    context.fill(
        Path(ellipseIn: ambientRect.insetBy(
            dx: size.width * 0.15, dy: size.height * 0.15
        )),
        with: .color(ambientGlow)
    )
}

--- 3f. Static overlay (reduceMotion) ---

private var plasmaStaticOverlay: some View {
    ZStack {
        // Static glow spots
        ForEach(0..<40, id: \.self) { i in
            Circle()
                .fill(
                    Color(
                        hue: 0.72 + Double(i % 10) * 0.02,
                        saturation: 0.85,
                        brightness: 0.65
                    ).opacity(0.12 + Double(i % 5) * 0.03)
                )
                .frame(
                    width: CGFloat(6 + (i % 8) * 3),
                    height: CGFloat(6 + (i % 8) * 3)
                )
                .position(
                    x: CGFloat(50 + (i * 37) % 300),
                    y: CGFloat(60 + (i * 53) % 600)
                )
        }

        // Ambient center glow
        Ellipse()
            .fill(
                Color(red: 0.85, green: 0.20, blue: 0.85).opacity(0.06)
            )
            .frame(width: 200, height: 200)
    }
}

private func renderPlasma(ctx: GraphicsContext, size: CGSize, t: Double) {
    for p in plasmaParticles {
        let driftX = sin(t * p.speed * 0.3 + p.phase) * p.driftX
        let driftY = cos(t * p.speed * 0.25 + p.phase * 1.3) * p.driftY

        let x = ((p.x + driftX).truncatingRemainder(dividingBy: 1.0) + 1.0)
            .truncatingRemainder(dividingBy: 1.0) * size.width
        let y = ((p.y + driftY).truncatingRemainder(dividingBy: 1.0) + 1.0)
            .truncatingRemainder(dividingBy: 1.0) * size.height

        let pulse = 0.5 + 0.5 * sin(t * (0.6 + p.speed * 0.4) + p.phase * 2.0)
        let particleSize = CGFloat(p.size * (0.6 + pulse * 0.9))
        let glowSize = particleSize * 2.5

        let breath = 0.4 + 0.6 * sin(t * (0.5 + p.speed * 0.3) + p.phase)
        let alpha = p.brightness * max(0.15, breath)

        let color = Color(
            hue: p.hue,
            saturation: 0.8 + 0.2 * sin(t * 0.2 + p.phase),
            brightness: 0.8 + 0.2 * breath
        ).opacity(alpha)

        // Glow ring
        let glowRect = CGRect(
            x: x - glowSize / 2, y: y - glowSize / 2,
            width: glowSize, height: glowSize
        )
        ctx.fill(Path(ellipseIn: glowRect), with: .color(color.opacity(alpha * 0.15)))

        // Core
        let coreRect = CGRect(
            x: x - particleSize / 2, y: y - particleSize / 2,
            width: particleSize, height: particleSize
        )
        ctx.fill(Path(ellipseIn: coreRect), with: .color(color))
    }
}
