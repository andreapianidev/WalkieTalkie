//  ThemeRegistry.swift

import SwiftUI

/// Central registry that aggregates all theme metadata from the 3 pack files.
/// Acts as a facade — Theme.swift calls `ThemeRegistry.metadata(for:)` to resolve
/// any Theme case into its full ThemeMetadata payload.
enum ThemeRegistry {
    /// All 19 themes registered once, resolved at first access.
    private static var registry: [Theme: ThemeMetadata] = {
        var dict: [Theme: ThemeMetadata] = [:]
        ThemeColorPack.register(into: &dict)
        ThemeIdentityPack.register(into: &dict)
        ThemeAnimatedPack.register(into: &dict)
        return dict
    }()

    /// Returns the full metadata for a given theme.
    /// - Parameter theme: The theme identifier.
    /// - Returns: The resolved ThemeMetadata.
    /// - Warning: Fatal error if the theme is not registered (should never happen
    ///   when all 3 packs are loaded and Theme enum stays in sync).
    static func metadata(for theme: Theme) -> ThemeMetadata {
        guard let meta = registry[theme] else {
            fatalError("ThemeRegistry: missing metadata for \(theme.rawValue)")
        }
        return meta
    }
}
