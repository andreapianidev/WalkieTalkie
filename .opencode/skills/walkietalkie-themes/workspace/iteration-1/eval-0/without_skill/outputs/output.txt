Full content of ThemeRegistry.swift:

//  ThemeRegistry.swift
//  WalkieTalkie

import Foundation

/// Registry centralizzato dei metadati tema. Popolato all'avvio dai pack file
/// (`ThemeColorPack`, `ThemeIdentityPack`, `ThemeAnimatedPack`) tramite il
/// pattern di registrazione `register(into:)`.
///
/// Uso: `ThemeRegistry.metadata(for: .tropical)`
enum ThemeRegistry {

    private static let storage: [Theme: ThemeMetadata] = {
        var dict = [Theme: ThemeMetadata]()
        ThemeColorPack.register(into: &dict)
        ThemeIdentityPack.register(into: &dict)
        ThemeAnimatedPack.register(into: &dict)
        return dict
    }()

    static func metadata(for theme: Theme) -> ThemeMetadata {
        guard let meta = storage[theme] else {
            // Fallback sicuro: ogni tema DEVE essere registrato.
            // Se manca, restituiamo il default (free) per evitare crash.
            assertionFailure("ThemeRegistry: '\(theme.rawValue)' non registrato in alcun pack")
            return storage[.defaultTheme]!
        }
        return meta
    }
}
